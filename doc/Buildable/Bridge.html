def mtu
  cmd = "sudo ip link show #{@name}"
  pp cmd
  Open3.popen3(cmd) do |stdin, stdout, stderr, wait_thr|
    exit unless wait_thr.value.success?
    stdout.readlines.join.match(/mtu (\d+)/)[1]
  end
end

def popen(cmd)
  Open3.popen3(cmd) do |stdin, stdout, stderr, wait_thr|
    puts stdout.readlines
    puts stderr.readlines
    exit unless wait_thr.value.success?
  end
end
